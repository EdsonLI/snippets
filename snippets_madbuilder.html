Bancos de dados

Abrir/fechar transação: 
<?php
TTransaction::open('mini_erp');
TTransaction::close();
?>
SQL manual: 
<?php
$conn = TTransaction::get();
$result = $conn->query('SELECT id FROM bank ORDER BY id');
$objects = $result->fetchAll(PDO::FETCH_CLASS, "stdClass");
if($objects) {
  foreach($objects as $object) { 
    echo $object->id;
  }    
}
?>
Inserir registro: 
<?php
$objeto = new Bank();
$objeto->campo = 'valor';
$objeto->store();
?>
Carregar registro pelo id: 
<?php $objeto = Bank::find( $id ); ?>
Carregar registro por outra coluna: 
<?php $objeto = Bank::where('coluna', '=', $valor)->first(); ?>
Alterar registro pelo id: 
<?php
$objeto = Bank::find( $id );
if($objeto) {
  $objeto->coluna = 'valor';
  $objeto->store();    
}
?>
Deletar registro pelo id: 
<?php
$objeto = Bank::find( $id );
if($objeto) {
  $objeto->delete();    
}
?>
Converter registro em vetor: 
<?php
$objeto = Bank::find( $id );
if($objeto) {
  $array = $objeto->toArray();    
}
?>
Converter registro em json: 
<?php
$objeto = Bank::find( $id );
if($objeto) {
  $array = $objeto->toJson();    
}
?>
Carregar registros através de filtros: 
<?php
$objs = Bank::where('coluna', '=', $valor)
          ->where('coluna2', '=', $valor2)
          ->orderBy('id')
          ->load();
?>
Carregar o primeiro registro encontrado: 
<?php
$objs = Bank::where('coluna',  '=', $valor)
          ->where('coluna2', '=', $valor2)
          ->orderBy('id')
          ->first();
?>
Carregar n registros: 
<?php
$objs = Bank::where('coluna',  '=', $valor)
          ->orderBy('id')
          ->take(10)
          ->load();
?>
Alterar múltiplos registros: 
<?php
Bank::where('id', 'IN', [1,2,3,4]) // id in (1,2,3,4)
    ->where('coluna1', '=', 'valor1') // coluna1 = 'valor1'
    ->set('coluna2', '50') // set coluna2 = '50'
    ->update();
?>
Deletar múltiplos registros: 
<?php
Bank::where('coluna', '=', $valor)
    ->where('coluna2', '=', $valor2)
    ->delete();
?>
Contar registros: 
<?php
Bank::where('coluna',  '=', $valor)
    ->where('coluna2', '=', $valor2)
    ->count();
?>
Carregar registros como vetor indexado: 
<?php
$array = Bank::where('coluna',  '=', $valor)
             ->where('coluna2', '=', $valor2)
             ->orderBy('id')
             ->getIndexedArray('primaryKey', 'coluna3');
?>
Habilitar logs de sql: 
<?php TTransaction::setLogger(new TLoggerSTD); ?>

  foreach($param['builder_datagrid_check'] as $check_id) {
     $bank = new Bank($check_id);

   }

Editar registros selecionados na listagem: 
<?php
if($param['builder_datagrid_check']) {
  foreach($param['builder_datagrid_check'] as $check_id) {
      $bank = new Bank($check_id);
      $bank->nome_da_coluna1 = 'novo valor';
      $bank->nome_da_coluna2 = 'novo valor';
      $bank->store();
  }

?>
Clonar registro: 
<?php
$bank = new Bank();
$bankClone = clone $bank;
unset($bankClone->id);
$bankClone->store();
?>

Campos

Habilitar campo: 
<?php TEntry::enableField(self::$formName, 'bank_id'); ?>
Desabilitar campo: 
<?php TEntry::disableField(self::$formName, 'bank_id'); ?>
Habilitar arrastar para reordenar valores de um TFieldList: 
<?php
$increaseFieldList = new TFieldList('increase_field_list');
$this->increaseFieldList = $increaseFieldList;
$this->increaseFieldList->enableSorting();
?>
Enviar dados para campo: 
<?php
$object = new stdClass();
$object->bank_id = 'insira o novo valor aqui';
//$object->fieldName = 'insira o novo valor aqui'; //sample
TForm::sendData(self::$formName, $object);
?>
Recarregar combo: 
<?php
$items = array();
$items['chave'] = 'valor';
TCombo::reload(self::$formName, 'stages_id', $items, true);
?>
Esconder campo: 
<?php TScript::create("$(\"[name='stages_id']\").closest('.fb-inline-field-container').hide()"); ?>
Mostrar campo: 
<?php TScript::create("$(\"[name='stages_id']\").closest('.fb-inline-field-container').show()"); ?>
Mostrar rótulo pelo texto: 
<?php TScript::create("$('label:contains(\"Título\")').show();"); ?>
Esconder rótulo pelo texto: 
<?php TScript::create("$('label:contains(\"Título\")').hide();"); ?>
Trocar rótulo pelo texto:

Trocar máscara do TEntry:
