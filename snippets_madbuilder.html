Bancos de dados

Abrir/fechar transação: 
<?php
TTransaction::open('mini_erp');
TTransaction::close();
?>
SQL manual: 
<?php
$conn = TTransaction::get();
$result = $conn->query('SELECT id FROM bank ORDER BY id');
$objects = $result->fetchAll(PDO::FETCH_CLASS, "stdClass");
if($objects) {
  foreach($objects as $object) { 
    echo $object->id;
  }    
}
?>
Inserir registro: 
<?php
$objeto = new Bank();
$objeto->campo = 'valor';
$objeto->store();
?>
Carregar registro pelo id: 
<?php $objeto = Bank::find( $id ); ?>
Carregar registro por outra coluna: 
<?php $objeto = Bank::where('coluna', '=', $valor)->first(); ?>
Alterar registro pelo id: 
<?php
$objeto = Bank::find( $id );
if($objeto) {
  $objeto->coluna = 'valor';
  $objeto->store();    
}
?>

Campos

Definir campo: 
<?php
$campo = new Campo('nome', 'string');
?>
Validar campo: 
<?php
$campo->validar('required');
?>
Obter valor do campo: 
<?php
$valor = $campo->getValor();
?>
Definir valor do campo: 
<?php
$campo->setValor('novo valor');
?>
Remover campo: 
<?php
$campo->remover();
?>
Listar campos: 
<?php
$campos = Campo::listar();
foreach ($campos as $campo) {
    echo $campo->getNome();
}
?>

SQL

SELECT simples: 
SELECT * FROM usuarios WHERE ativo = 1;
INSERT: 
INSERT INTO usuarios (nome, email) VALUES ('João', 'joao@email.com');
UPDATE: 
UPDATE usuarios SET ativo = 0 WHERE id = 5;
DELETE: 
DELETE FROM usuarios WHERE id = 10;
