Bancos de dados

Abrir/fechar transação: 
<?php
   TTransaction::open('mini_erp');
   TTransaction::close();
?>
SQL manual: 
<?php
   $conn = TTransaction::get();
   $result = $conn->query('SELECT id FROM bank ORDER BY id');
   $objects = $result->fetchAll(PDO::FETCH_CLASS, "stdClass");
   if($objects) {
      foreach($objects as $object) { 
         echo $object->id;
      }    
   }
?>
Inserir registro: 
<?php
   $objeto = new Bank();
   $objeto->campo = 'valor';
   $objeto->store();
?>
Carregar registro pelo id: 
<?php $objeto = Bank::find( $id ); ?>
Carregar registro por outra coluna: 
<?php $objeto = Bank::where('coluna', '=', $valor)->first(); ?>
Alterar registro pelo id: 
<?php
   $objeto = Bank::find( $id );
   if($objeto) {
      $objeto->coluna = 'valor';
      $objeto->store();    
   }
?>
Deletar registro pelo id: 
<?php
   $objeto = Bank::find( $id );
   if($objeto) {
      $objeto->delete();    
   }
?>
Converter registro em vetor: 
<?php
   $objeto = Bank::find( $id );
   if($objeto) {
      $array = $objeto->toArray();    
   }
?>
Converter registro em json: 
<?php
   $objeto = Bank::find( $id );
   if($objeto) {
      $array = $objeto->toJson();    
   }
?>
Carregar registros através de filtros: 
<?php
   $objs = Bank::where('coluna', '=', $valor)
               ->where('coluna2', '=', $valor2)
               ->orderBy('id')
               ->load();
?>
Carregar o primeiro registro encontrado: 
<?php
   $objs = Bank::where('coluna',  '=', $valor)
               ->where('coluna2', '=', $valor2)
               ->orderBy('id')
               ->first();
?>
Carregar n registros: 
<?php
   $objs = Bank::where('coluna',  '=', $valor)
               ->orderBy('id')
               ->take(10)
               ->load();
?>
Alterar múltiplos registros: 
<?php
   Bank::where('id', 'IN', [1,2,3,4])
       ->where('coluna1', '=', 'valor1')
       ->set('coluna2', '50')
       ->update();
?>
Deletar múltiplos registros: 
<?php
   Bank::where('coluna', '=', $valor)
       ->where('coluna2', '=', $valor2)
       ->delete();
?>
Contar registros: 
<?php
   Bank::where('coluna',  '=', $valor)
       ->where('coluna2', '=', $valor2)
       ->delete();
?>
Carregar registros como vetor indexado: 
<?php
   $array = Bank::where('coluna',  '=', $valor)
                ->where('coluna2', '=', $valor2)
                ->orderBy('id')
                ->getIndexedArray('primaryKey', 'coluna3');
?>
Habilitar logs de sql: 
<?php TTransaction::setLogger(new TLoggerSTD); ?>
